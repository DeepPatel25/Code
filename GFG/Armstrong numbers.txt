class Solution {
  public:
    string armstrongNumber(int n){
        long long ans = 0;
        int extra = n;
        
        while(n != 0) {
            int temp = n % 10;
            ans += (temp * temp * temp);
            n /= 10;
        }
        
        if(extra == ans) return "Yes";
        return "No";
    }
};

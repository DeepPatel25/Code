//User function template for C++
class Solution{
public:
	
	bool fascinating(int n) {
	    string str = to_string(n) + to_string(n * 2) + to_string(n * 3);
	    sort(str.begin(), str.end());
	    
	    return str == "123456789";
	}
};

//{ Driver Code Starts.

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        Solution ob;
        auto ans = ob.fascinating(n);
        if (ans) {
            cout << "Fascinating\n";
        } else {
            cout << "Not Fascinating\n";
        }
    }
    return 0;
}

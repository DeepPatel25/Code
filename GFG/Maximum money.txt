class Solution {
  public:
    int maximizeMoney(int N , int K) {
        return (K * (N / 2 + N % 2));
    }
};

    while (t--) {
        int N,K;

        cin>>N>>K;

        Solution ob;
        cout << ob.maximizeMoney(N,K) << endl;
    }
    return 0;
}
